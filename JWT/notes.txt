# Create a RSA certificate for signing JWT:

    openssl req -x509 -nodes -newkey rsa:2048 -keyout JWT/JWT-RS256-2048.key -out JWT/JWT-RS256-2048.cert -subj "/CN=JWTProvider"


# SAP HANA:

## Create a JWT Provider

    CREATE JWT PROVIDER JWTPROVIDER WITH ISSUER "https://jwtprovider/" CLAIM "username" AS EXTERNAL IDENTITY;

## Create a PSE Collection

    CREATE PSE JWTPSE;

## Associate the PSE Collection with the JWT Provider

    SET PSE JWTPSE PURPOSE JWT FOR PROVIDER JWTPROVIDER;

## Create the Certificate

    CREATE CERTIFICATE FROM 'XXXX' COMMENT 'Comment';

## Add the Certificate to the PSE

    ALTER PSE JWTPSE ADD CERTIFICATE 162543;
